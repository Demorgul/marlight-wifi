---------                ------------------------------------------------------

0x01, 0x55        Включить все 4 канала
0x02, 0x55        Выключить все 4 канала



0x03, 0x55        Увеличить текущий уровень яркости
0x04, 0x55        Уменьшить текущий уровень яркости

0x05, 0x55        Сделать "холоднее" цветовую температуру
0x06, 0x55        Сделать "теплее" цветовую температуру

0x07, 0x55        Выставить цветовую температуру и яркость "по умолчанию"



0x08, 0x55        Включить канал №1
0x09, 0x55        Выключить канал №1

0x0A, 0x55        Включить канал №2
0x0B, 0x55        Выключить канал №2

0x0C, 0x55        Включить канал №3
0x0D, 0x55        Выключить канал №3

0x0E, 0x55        Включить канал №4
0x0F, 0x55        Выключить канал №4


----------------------- РЕЖИМ RGB -----------------------

0x12, 0x55                 Переключиться в режим работы RGB
0x10, 0x55                 Уменьшить яркость в режиме RGB
0x11, 0x55                 Увеличить яркость в режиме RGB
0x13, R, G, B, 0x55     Установить цвет в режиме RGB, где R,G,B яркость по каналам RGB


----------------------- ПРЕДУСТАНОВЛЕННЫЕ РЕЖИМЫ -----------------------

0x14, 0x55                Переключиться в режим НОЧЬ
0x15, 0x55                Переключиться в режим СОВЕЩАНИЕ
0x16, 0x55                Переключиться в режим ЧТЕНИЕ
0x17, 0x55                Режим какого-то одного цвета или плавной смены палитры (режимы переключаются по кругу при передаче этой посылки))

0x18, H1, M1, H2, M2, 0x09, 0x14, 0x55     Режим Таймер, где H1:M1 - время включения, H2:M2 - время выключения
0x19, H1, M1, 0x09, 0x14, 0x55        Режим Будильник, где H1:M1 - время включения
0x1A, 0x55                Режим СОН (как я понял через какое-то время выключает лампочку, правда не выяснил как сразу резко или постепенно снижая яркость)
0x1B, R, G, B, 0x55            Режим RECREATION включения заранее заданного цвета 
                    (дело в том что при долгом "тапе" в программе на телефонах можно выбрать цвет и программа его запоминает, 
                    а потом каждый раз при коротком нажатии передает, получается что то вроде одновременного посыла 2 команд - перейти в режим RGB и сразу выставить палитру, только за одну команду)
